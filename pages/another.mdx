export const Counter = () => {
  const [count, setCount] = useState(0);
  return (
    <div>
      <button onClick={() => setCount(count + 1)} className="border-[1px] border-solid border-white mt-3 py-[2px] px-[6px]">
        Clicked {count} times
      </button>
    </div>
  );
};

<Counter />
